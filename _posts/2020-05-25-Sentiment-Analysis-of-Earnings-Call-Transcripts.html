#write a function that breaks the transcripts into list of words
def create_transcript_wordlist(filename):
    with open(filename, encoding='latin-1') as file:
        word_list = []
        for line in file:
            line = line.strip()
            if line:
                words = line.split()
                for each in words:
                    word_list.append(each)
    return (word_list)

# Write a function that calculates the total of sentiment score and gives other important information about
#positive and negative words score from all the words in the transcript
def sum_of_sentiment_score(word_list, sentiment_dict):
    '''takes the word list of the transcripts and checks each word in the sentiment dictionary and calculates the sentiment
    score for the entire transcript words'''
    incl_word_count = 0
    excl_word_count = 0
    sum_of_score = 0
    sum_pos_score = 0
    sum_neg_score = 0
    pos_sent_list = []
    neg_sent_list = []
    incl_word_list = []
    excl_word_list = []
    for each in word_list:
        if each in sentiment_dict.keys():
            incl_word_list.append(each)
            incl_word_count += 1
            each_score = sentiment_dict[each]
            sum_of_score += each_score
            if each_score > 0:
                pos_sent_list.append(each)
                sum_pos_score += each_score
            else:
                neg_sent_list.append(each)
                sum_neg_score += each_score

        if each not in sentiment_dict.keys():
            excl_word_list.append(each)
            excl_word_count += 1
    pos_word_count = len(pos_sent_list)
    neg_word_count = len(neg_sent_list)
    #print(sum_of_score,inlist_count,outlist_count)

    return sum_of_score,incl_word_count, excl_word_count,pos_word_count,neg_word_count,sum_pos_score,sum_neg_score

#write a function to create pointplot 
def create_pointplot(dataframe, col1,col2):
    plt.figure(figsize=[14.70, 8.27])
    ax = plt.axes()
    sb.pointplot(data=dataframe,x=col1,y=col2,ax=ax);
    plt.ylabel('Average Sentiment Score of transcript'.title(),fontsize = 12, weight = "bold")
    plt.xlabel('Financial Quarters'.title(),fontsize = 12, weight = "bold")
